    <script>
    document.getElementById('submit').addEventListener('click', () => {

    const submitButton = document.getElementById('submit');
    const spinner = document.getElementById('spinner');
    const buttonText = document.getElementById('button-text');    const errorBanner = document.getElementById('invalid-form');
    errorBanner.hidden = true;

    spinner.classList.remove('d-none');
    buttonText.textContent = "Generating…";
    submitButton.disabled = true;

    let formData = new FormData();

    formData.append('fname', document.getElementById('fname').value);
    formData.append('lname', document.getElementById('lname').value);
    formData.append('email', document.getElementById('email').value);
    formData.append('occupation', document.getElementById('occupation').value);
    formData.append('genre', document.getElementById('genre').value);
    formData.append('length', document.getElementById('length').value);
    formData.append('format', document.getElementById('format').value);
    formData.append('age', document.getElementById('age').value);
    formData.append('outputLength', document.getElementById('outputLength').value);


    fetch('http://127.0.0.1:3000', { method: 'POST', body: formData })
        .then(async response => {
            let json = await response.json();
            return { status: response.status, json };
        })
        .then(({ status, json }) => {

            const errorBanner = document.getElementById('invalid-form');
            const errorMessages = document.getElementsByClassName('text-danger');

            // Clear all old errors
            errorBanner.hidden = true;
            for (let el of errorMessages) {
                el.innerHTML = '&nbsp;';
            }

             spinner.classList.add('d-none');

            if (status === 400) {
                errorBanner.innerText = 'Form has errors. Please correct them and resubmit.';
                errorBanner.hidden = false;
                errorBanner.classList.remove('d-none');

                for (let error of json.errors) {
                    const errorId = `${error.path}-error`;
                    document.getElementById(errorId).innerHTML = error.msg;
                }

                spinner.classList.add('d-none');
                buttonText.textContent = "Submit";
                submitButton.disabled = false;

                return;
            }

            document.getElementById('ai-output').innerText = json.text;
        })
        .catch(err => console.error('Fetch Error:', err));
});

   
    </script>
